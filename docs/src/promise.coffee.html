checkCallbacks = (arrayOfCallbacks) ->
  i = arrayOfCallbacks.length
  
  while i
    arg = arrayOfCallbacks[--i]
    if arg? and typeof arg != "function"
      throw new Error 'callback is not a function'

resolved = (value) ->
  p = {}
  
  p.then = (callback) ->
    checkCallbacks arguments
    try
      if callback
        nextValue = callback value
        return promise( nextValue? ? nextValue : value )
    catch e
      return rejected e

  p

rejected = (reason) ->
  p = {}
  
  p.then = (callback, errback) ->
    checkCallbacks arguments
    
    try
      if errback
        nextValue = errback reason
        return Promise.promise( nextValue? ? nextValue : reason )
      return rejected reason
    catch e
      return rejected e

  p

  listeners = []
  progressHandlers = []

    checkCallbacks arguments
    
    deferred = Promise.defer()

      promise.then(callback, errback)

    deferred.promise

      throw new Error "already completed"

isPromise = (promiseOrValue) ->
  if typeof promiseOrValue != "object"
    return false
  if promiseOrValue.then? and typeof promiseOrValue.then == "function"
    return true

_when = (promiseOrValue, callback, errback, progressHandler) ->
  trustedPromise = promise promiseOrValue
  
  trustedPromise.then callback, errback, progressHandler
  
promise = (promiseOrValue) ->
  deferred = defer()
  
  if isPromise promiseOrValue
    promiseOrValue.then deferred.resolve, deferred.reject, deferred.progress
    promise = deferred.promise
  else
    deferred.resolve promiseOrValue
    promise = deferred.promise
    
  promise
